CREATE TABLE Users
	(location CHAR(20),
	name CHAR(20),
	uid  CHAR(11),
	desc CHAR(10),
	cdate CHAR(10,
	PRIMARY KEY (uid))
CREATE TABLE UBadges
	(bdate CHAR(10),
	uid CHAR(11) NOT NULL,
	class CHAR(2) NOT NULL
	PRIMARY KEY(bdate, uid),
	FOREIGN KEY (uid) REFRENCES Users
		ON DELETE CASCADE,
	FOREIGN KEY (class) REFRENCES Badges)	
CREATE TABLE Badges
	(class CHAR(2),
	name CHAR(11)
	PRIMARY KEY (class))
CREATE TABLE votes
	(vType CHAR(5),
	vdate CHAR(10),
	vno CHAR(11),
	uid CHAR(11) NOT NULL,
	pid CHAR(11) NOT NULL,
	PRIMARY KEY(vno, pid),
	FOREIGN KEY (pid) REFRENCES Posts
		ON DELETE CASCADE,
	FOREIGN KEY (uid) REFRENCE Users)
Create TABLE comment
	(pid CHAR (6),
	uid CHAR (11),
	cdate CHAR(10),
	ctetxt CHAR(20),
	PRIMARY KEY(pid, uid),
	FOREIGN KEY (pid) REFRENCES Posts,
	FOREIGN KEY (uid) REFRENCES Users)
CREATE TABLE Posts
	(pdate CHAR(11),
	pid CHAR(6),
	title CHAR(30),
	body CHAR(120),
	uid CHAR(11) NOT NULL,
	PRIMARY KEY(pid)
	FOREIGN KEY (uid) REFRENCE Users
		ON DELETE NO ACTION)
CREATE TABLE Questions
	(pid CHAR(6),
	apid CHAR(6),
	PRIMARY KEY(pid),
	FOREIGN KEY(pid) REFRENCES Posts
		ON DELETE CASCADE
	FOREIGN KEY (apid) REFRENCES Answers)
CREATE TABLE Answers
	(pid CHAR(6),
	qpid CHAR(6),
	PRIMARY KEY(pid)
	FOREIGN KEY(pid) REFRENCES Posts
		ON DELETE CASCADE
	FOREIGN KEY(qpid) REFRENCES Questions
		ON DELETE NO ACTION)
